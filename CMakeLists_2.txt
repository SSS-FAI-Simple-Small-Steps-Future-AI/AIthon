cmake_minimum_required(VERSION 3.20)
project(PyVMLang VERSION 0.1.0 LANGUAGES CXX C)

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_EXPORT_COMPILE_COMMANDS ON)

# Build type
if(NOT CMAKE_BUILD_TYPE)
    set(CMAKE_BUILD_TYPE Release)
endif()

set(CMAKE_CXX_FLAGS_DEBUG "-g -O0 -DDEBUG")
set(CMAKE_CXX_FLAGS_RELEASE "-O3 -DNDEBUG")

# Find LLVM 21
find_package(LLVM 21 REQUIRED CONFIG)

message(STATUS "Found LLVM ${LLVM_PACKAGE_VERSION}")
message(STATUS "Using LLVMConfig.cmake in: ${LLVM_DIR}")

include_directories(${LLVM_INCLUDE_DIRS})
separate_arguments(LLVM_DEFINITIONS_LIST NATIVE_COMMAND ${LLVM_DEFINITIONS})
add_definitions(${LLVM_DEFINITIONS_LIST})

# Find Python3 for AST parsing
find_package(Python3 REQUIRED COMPONENTS Interpreter Development)

# Include directories
include_directories(
    ${CMAKE_SOURCE_DIR}/include
    ${Python3_INCLUDE_DIRS}
)

# Compiler sources
set(COMPILER_SOURCES
    src/main.cpp
    src/compiler/compiler.cpp
    src/ast/ast_nodes.cpp
    src/ast/python_ast_converter.cpp
    src/codegen/llvm_codegen.cpp
    src/codegen/async_transformer.cpp
    src/codegen/async_to_actor.cpp
    src/validator/project_validator.cpp
)

# Runtime sources
set(RUNTIME_SOURCES
    src/runtime/actor_process.cpp
    src/runtime/scheduler.cpp
    src/runtime/heap.cpp
    src/runtime/pyobject.cpp
    src/runtime/exceptions.cpp
    src/runtime/green_threads.cpp
    src/runtime/actor_gc.cpp
)

# LLVM components we need
llvm_map_components_to_libnames(llvm_libs
    core
    irreader
    support
    native
    orcjit
    passes
    target
    transformutils
    analysis
    ipo
    instcombine
    scalaropts
    vectorize
)

# Main compiler executable
add_executable(pyvm_compiler ${COMPILER_SOURCES} ${RUNTIME_SOURCES})

target_link_libraries(pyvm_compiler
    ${llvm_libs}
    ${Python3_LIBRARIES}
    pthread
)

# Runtime shared library
add_library(pyvm_runtime SHARED
    runtime_lib/runtime.cpp
    ${RUNTIME_SOURCES}
)

target_link_libraries(pyvm_runtime pthread)

# Tests
enable_testing()
add_subdirectory(tests)

# Install targets
install(TARGETS pyvm_compiler DESTINATION bin)
install(TARGETS pyvm_runtime DESTINATION lib)
